Real Spherical Harmonics times r^l based on wikipedia page https://en.wikipedia.org/wiki/Spherical_harmonics.
Run this cell, and it will define all the needed functions, rlY = r^l Y^l_m(x,y,z).
The Soherical Harmonics will return 0 if a invalid input is given for example l=3, m=5.

(%i18) Am( m, x, y) : = sum( binomial( m, p)· x ^ p· y ^( m − p)· cos(( m − p)· %pi / 2), p, 0, m) ;
Bm( m, x, y) : = sum( binomial( m, p)· x ^ p· y ^( m − p)· sin(( m − p)· %pi / 2), p, 0, m) ;
BigPi( l, m, x, y, z) : = (( l − m) ! /( l + m) !) ^( 1 / 2)· sum(( − 1) ^ k· 2 ^( − l)· binomial( l, k)· binomial( 2· l − 2· k, l)·( l − 2· k) ! /( l − 2· k − m) !·( x ^ 2 + y ^ 2 + z ^ 2) ^ k· z ^( l − 2· k − m), k, 0, floor(( l − m) / 2)) ;
BigPiNoConst( l, m, x, y, z) : =                       sum(( − 1) ^ k· 2 ^( − l)· binomial( l, k)· binomial( 2· l − 2· k, l)·( l − 2· k) ! /( l − 2· k − m) !·( x ^ 2 + y ^ 2 + z ^ 2) ^ k· z ^( l − 2· k − m), k, 0, floor(( l − m) / 2)) ;

rlYlpm( l, m, x, y, z) : = sqrt(( 2· l + 1) / 2 / %pi)· BigPi( l, m, x, y, z)· Am( m, x, y) ;
rlYlnm( l, m, x, y, z) : = sqrt((( 2· l + 1) / 2) / %pi)· BigPi( l, m, x, y, z)· Bm( m, x, y) ;
rlY0( l, x, y, z) : = sqrt((( 2· l + 1) / 4) / %pi)· BigPi( l, 0, x, y, z) ;

rlYlpmNoConst( l, m, x, y, z) : = BigPiNoConst( l, m, x, y, z)· Am( m, x, y) ;
rlYlnmNoConst( l, m, x, y, z) : = BigPiNoConst( l, m, x, y, z)· Bm( m, x, y) ;
rlY0NoConst( l, x, y, z) : = BigPiNoConst( l, 0, x, y, z) ;

rlYpre( l, m, x, y, z) : =
block([],
if m> 0
   then rlYlpm( l, m, x, y, z)
else if m = 0
   then rlY0( l, x, y, z)
else
    rlYlnm( l, − m, x, y, z)
   ) ;

rlY( l, m, x, y, z) : =
block([], if l< abs( m)
   then 0
  else
     rlYpre( l, m, x, y, z)
) ;

rlYpreNoConst( l, m, x, y, z) : =
block([],
if m> 0
   then rlYlpmNoConst( l, m, x, y, z)
else if m = 0
   then rlY0NoConst( l, x, y, z)
else
    rlYlnmNoConst( l, − m, x, y, z)
   ) ;

rlYNoConst( l, m, x, y, z) : =
block([], if l< abs( m)
   then 0
  else
     rlYpreNoConst( l, m, x, y, z)
) ;
